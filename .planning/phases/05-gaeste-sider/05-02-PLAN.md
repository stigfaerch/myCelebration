3. Modify `src/app/[uuid]/page.tsx` (replace existing skeleton):
   ```tsx
   import { RichTextDisplay } from '@/components/admin/RichTextDisplay'
   import { resolveGuest } from '@/lib/auth/resolveGuest'
   import { getFestInfo } from '@/lib/actions/information'
   import { getMyInvitation } from '@/lib/actions/guest/invitations'
   import { getChoiceDefinitions, getMyChoiceAnswers } from '@/lib/actions/guest/choices'
   import { getMyPerformances } from '@/lib/actions/guest/performances'
   import { getAssignmentCount } from '@/lib/actions/guest/tasks'
   // components...

   export default async function ForsidePage({ params }: { params: Promise<{ uuid: string }> }) {
     const { uuid } = await params
     const guest = await resolveGuest()
     const [festInfo, invitation, definitions, answers, performances, taskCount] = await Promise.all([
       getFestInfo(),
       guest.type === 'screen' ? null : getMyInvitation(),
       getChoiceDefinitions(),
       guest.type === 'screen' ? [] : getMyChoiceAnswers(),
       guest.type === 'screen' ? [] : getMyPerformances(),
       guest.type === 'screen' ? 0 : getAssignmentCount(),
     ])

     return (
       <div className="p-4 space-y-6">
         {/* Festinfo */}
         {festInfo?.description && <section><RichTextDisplay content={festInfo.description} /></section>}

         {/* Screen-type: only show festinfo */}
         {guest.type !== 'screen' && (
           <>
             <InvitationAccept initial={invitation} />
             <TaskIndicator uuid={uuid} count={taskCount} />
             <ChoiceAnswers definitions={definitions} initialAnswers={answers} />
             <PerformanceManager initialPerformances={performances} />
           </>
         )}
       </div>
     )
   }
   ```
