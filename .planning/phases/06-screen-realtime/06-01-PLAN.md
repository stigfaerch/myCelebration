   ```ts
       realtime: {

         },
       },
     })

   }
   ```

<task id="2" title="accept_swap_request PL/pgSQL function (migration 004)">
<action>
Create `supabase/migrations/004_swap_accept_rpc.sql`:

```sql
-- myCelebration — Atomic swap-request acceptance
-- Apply after 001_initial_schema.sql, 002_rls.sql, 003_program_items_swap_rpc.sql
--
-- Semantics:
--   First caller to successfully lock and update the swap wins. Subsequent
--   callers find status != 'pending' and raise 'already_accepted'. The two
--   affected task_assignments are swapped atomically under the same lock.
--
-- Arguments:
--   p_swap_id          — swap_requests.id
--   p_accepter_guest_id — the guest accepting (must own an assignment for a
--                         task in the swap's desired_task_ids)
--   p_accepter_task_id — the specific task_id the accepter is giving up
--                         (must be in desired_task_ids AND currently assigned
--                         to p_accepter_guest_id)
--
-- Returns: nothing (raises on failure)
--
-- Failure modes:
--   - raise 'already_accepted' if status != 'pending'
--   - raise 'invalid_target' if p_accepter_task_id not in desired_task_ids
--   - raise 'not_assigned' if accepter does not own an assignment for
--     p_accepter_task_id
--   - raise 'self_accept' if accepter is the requester
--   - raise 'not_found' if swap doesn't exist

create or replace function accept_swap_request(
  p_swap_id uuid,
  p_accepter_guest_id uuid,
  p_accepter_task_id uuid
)
returns void
language plpgsql
security definer
set search_path = public
as $$
declare
  v_swap record;
  v_requester_assignment record;
  v_accepter_assignment record;
begin
  -- Lock the swap row to serialize concurrent acceptors
  select id, requester_assignment_id, desired_task_ids, status
    into v_swap
    from swap_requests
    where id = p_swap_id
    for update;

  if not found then
    raise exception 'not_found';
  end if;

  if v_swap.status <> 'pending' then
    raise exception 'already_accepted';
  end if;

  -- Validate the accepter's chosen task is in the desired list
  if not (p_accepter_task_id = any(v_swap.desired_task_ids)) then
    raise exception 'invalid_target';
  end if;

  -- Lock the requester's assignment row (their current task)
  select id, task_id, guest_id, is_owner
    into v_requester_assignment
    from task_assignments
    where id = v_swap.requester_assignment_id
    for update;

  if not found then
    raise exception 'not_found';
  end if;

  -- Reject self-accept (requester trying to accept their own swap)
  if v_requester_assignment.guest_id = p_accepter_guest_id then
    raise exception 'self_accept';
  end if;

  -- Lock the accepter's assignment row for the task they're giving up
  select id, task_id, guest_id, is_owner
    into v_accepter_assignment
    from task_assignments
    where task_id = p_accepter_task_id
      and guest_id = p_accepter_guest_id
    for update;

  if not found then
    raise exception 'not_assigned';
  end if;

  -- Perform the swap: transfer ownership of the two assignments
  --   - requester_assignment moves to accepter_guest
  --   - accepter_assignment moves to requester_guest
  --
  -- `is_owner` is preserved on each task (it describes the original owner
  -- of the task, not the current assignee — see Phase 3 design note).

  update task_assignments
    set guest_id = p_accepter_guest_id
    where id = v_requester_assignment.id;

  update task_assignments
    set guest_id = v_requester_assignment.guest_id
    where id = v_accepter_assignment.id;

  -- Mark the swap as accepted
  update swap_requests
    set status = 'accepted'
    where id = v_swap.id;
end;
$$;

grant execute on function accept_swap_request(uuid, uuid, uuid)
  to anon, authenticated, service_role;
```

Verification notes to record in SUMMARY:
- The `unique(task_id, guest_id)` constraint on `task_assignments` (migration 001 line 166) guarantees no accepter can hold two assignments for the same task. The update statements above never violate this constraint because we're swapping guest_ids between two rows that already exist.
- `is_owner` semantics: Phase 3 set `is_owner=true` for the original assignee only. Post-swap, neither side's `is_owner` flag is changed — the "byttet til"/"oprindelig" badge in TaskList (Phase 5) correctly reflects the pre-swap ownership.

```sql
-- myCelebration — RLS policies for browser realtime subscriptions
-- Apply after 002_rls.sql
--
-- Subscription surface (all SELECT-only via anon key):
--   screen_state       — already permitted by 002_rls.sql (screen_state_realtime_select)
--   swap_requests      — guests need to see swaps targeting their tasks
--   task_assignments   — needed to derive "is this swap's desired_task_ids
--                        intersecting with MY tasks?" — the browser client
--                        cannot query this at subscription-setup time without
--                        an anon SELECT policy.
--
-- Mutations remain service-role-only — no INSERT/UPDATE/DELETE policies added here.
--
-- Privacy note: swap_requests and task_assignments are not considered sensitive
-- within the party — the existing shared-password cookie already gates access
-- to the app's entire UI. This is defense-in-depth against accidental exposure
-- via the realtime API, not a confidentiality boundary.

create policy "swap_requests_realtime_select"
  on swap_requests
  for select
  using (true);

create policy "task_assignments_realtime_select"
  on task_assignments
  for select
  using (true);

-- Note: we deliberately do NOT open SELECT on:
--   - guests (private info: emails, phones, relations)
--   - performances (admin curation state)
--   - choice_definitions / guest_choices (party configuration + personal answers)
--   - memories / photos (content belongs to guests; gallery read goes via server action)
-- The 06-02 components resolve any guest/task display data server-side when
-- rendering the initial page, then update via realtime payloads that only
-- contain the three SELECT-opened tables above.
```
