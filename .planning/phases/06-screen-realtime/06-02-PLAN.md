   ```tsx
   'use client'
   import * as React from 'react'
   import { useRouter } from 'next/navigation'
   import { getSupabaseBrowserClient } from '@/lib/supabase/client'

   interface Props {
     guestId: string
     children: React.ReactNode
   }

   export function ScreenRouter({ guestId, children }: Props) {
     const router = useRouter()

     React.useEffect(() => {
       const client = getSupabaseBrowserClient()
       const channel = client
         .channel(`screen_state:${guestId}`)
         .on(
           'postgres_changes',
           {
             event: '*',
             schema: 'public',
             table: 'screen_state',
             filter: `guest_id=eq.${guestId}`,
           },
           () => {
             router.refresh()
           }
         )
         .subscribe()

       return () => {
         client.removeChannel(channel)
       }
     }, [guestId, router])

     return <>{children}</>
   }
   ```
   - The router.refresh re-runs the server component which reads the fresh `screen_state` row + referenced content. No payload deserialization needed.

   ```ts
   export interface IncomingSwapRequest {
     id: string
     requester_assignment_id: string
     desired_task_ids: string[]
     status: SwapStatus
     created_at: string
     // Resolved display fields
     requester_name: string
     requester_task_title: string
     // Which of MY task_ids are eligible targets for this swap
     my_matching_task_ids: string[]
   }

   export async function getIncomingSwapRequests(): Promise<IncomingSwapRequest[]> {
     const guest = await resolveGuest()

     // 1. My task_ids
     const { data: mine, error: mineErr } = await supabaseServer
       .from('task_assignments')
       .select('task_id')
       .eq('guest_id', guest.id)
     if (mineErr) throw new Error('Failed to load assignments')
     const myTaskIds = ((mine ?? []) as Array<{ task_id: string }>).map((r) => r.task_id)
     if (myTaskIds.length === 0) return []

     // 2. All pending swap_requests where desired_task_ids OVERLAPS my tasks
     //    Postgres array overlap: .overlaps('desired_task_ids', myTaskIds) per PostgREST
     const { data: swaps, error: swapsErr } = await supabaseServer
       .from('swap_requests')
       .select(
         'id, requester_assignment_id, desired_task_ids, status, created_at'
       )
       .eq('status', 'pending')
       .overlaps('desired_task_ids', myTaskIds)
       .order('created_at', { ascending: false })
     if (swapsErr) throw new Error('Failed to load incoming swap requests')

     const swapRows = (swaps ?? []) as Array<{
       id: string
       requester_assignment_id: string
       desired_task_ids: string[]
       status: SwapStatus
       created_at: string
     }>

     // Filter out self-originated (assignment owned by current guest)
     // and resolve display info via a second query
     if (swapRows.length === 0) return []

     const requesterAssignmentIds = swapRows.map((s) => s.requester_assignment_id)
     const { data: assignments, error: assignErr } = await supabaseServer
       .from('task_assignments')
       .select('id, guest_id, task_id, tasks(title), guests(name)')
       .in('id', requesterAssignmentIds)
     if (assignErr) throw new Error('Failed to resolve swap requester details')

     const assignmentMap = new Map(
       ((assignments ?? []) as Array<Record<string, unknown>>).map((a) => [
         a.id as string,
         a,
       ])
     )

     const mySet = new Set(myTaskIds)

     return swapRows
       .filter((s) => {
         const a = assignmentMap.get(s.requester_assignment_id)
         return a && (a.guest_id as string) !== guest.id
       })
       .map((s) => {
         const a = assignmentMap.get(s.requester_assignment_id)!
         const tasks = a.tasks as { title: string } | Array<{ title: string }> | null
         const guests_ = a.guests as { name: string } | Array<{ name: string }> | null
         const taskTitle = Array.isArray(tasks) ? tasks[0]?.title : tasks?.title
         const guestName = Array.isArray(guests_) ? guests_[0]?.name : guests_?.name
         return {
           id: s.id,
           requester_assignment_id: s.requester_assignment_id,
           desired_task_ids: s.desired_task_ids,
           status: s.status,
           created_at: s.created_at,
           requester_name: guestName ?? 'Ukendt',
           requester_task_title: taskTitle ?? 'Ukendt opgave',
           my_matching_task_ids: s.desired_task_ids.filter((id) => mySet.has(id)),
         }
       })
   }
   ```

   Add `acceptSwapRequest(swapId, accepterTaskId)`:
   ```ts
   export async function acceptSwapRequest(
     swapId: string,
     accepterTaskId: string
   ): Promise<void> {
     const guest = await assertNotScreen()

     if (!UUID_RE.test(swapId) || !UUID_RE.test(accepterTaskId)) {
       throw new Error('Ugyldigt valg')
     }

     const { error } = await supabaseServer.rpc('accept_swap_request', {
       p_swap_id: swapId,
       p_accepter_guest_id: guest.id,
       p_accepter_task_id: accepterTaskId,
     })

     if (error) {
       // Map known PL/pgSQL errors to user-facing Danish messages
       const msg = error.message || ''
       if (msg.includes('already_accepted')) {
         throw new Error('Denne bytte-forespørgsel er allerede accepteret')
       }
       if (msg.includes('invalid_target')) {
         throw new Error('Ugyldigt valg')
       }
       if (msg.includes('not_assigned')) {
         throw new Error('Du har ikke den valgte opgave')
       }
       if (msg.includes('self_accept')) {
         throw new Error('Du kan ikke acceptere din egen forespørgsel')
       }
       if (msg.includes('not_found')) {
         throw new Error('Bytte-forespørgsel ikke fundet')
       }
       throw new Error('Kunne ikke acceptere bytte-forespørgsel')
     }

     revalidatePath('/' + guest.id + '/opgaver')
   }
   ```

2. Create `src/components/guest/IncomingSwapList.tsx` (client):
   ```tsx
   'use client'
   import * as React from 'react'
   import { useRouter } from 'next/navigation'
   import { Button } from '@/components/ui/button'
   import { getSupabaseBrowserClient } from '@/lib/supabase/client'
   import { acceptSwapRequest } from '@/lib/actions/guest/tasks'
   import type { IncomingSwapRequest } from '@/lib/actions/guest/tasks'

   interface Props {
     initialIncoming: IncomingSwapRequest[]
     taskTitleMap: Record<string, string> // my task_id → title, for the "give up" selector
   }

   export function IncomingSwapList({ initialIncoming, taskTitleMap }: Props) {
     const router = useRouter()
     const [incoming, setIncoming] = React.useState(initialIncoming)
     const [prev, setPrev] = React.useState(initialIncoming)
     const [selectedMap, setSelectedMap] = React.useState<Record<string, string>>({})
     const [pending, setPending] = React.useState<Set<string>>(new Set())
     const [errors, setErrors] = React.useState<Record<string, string>>({})

     // Resync from server-provided initial data
     if (prev !== initialIncoming) {
       setPrev(initialIncoming)
       setIncoming(initialIncoming)
     }

     // Realtime: any swap_requests change → router.refresh (lets server re-compute
     // the filtered set with full owner resolution). Use filter on status only
     // since desired_task_ids overlap can't be expressed as a realtime filter.
     React.useEffect(() => {
       const client = getSupabaseBrowserClient()
       const channel = client
         .channel('incoming_swaps')
         .on(
           'postgres_changes',
           { event: '*', schema: 'public', table: 'swap_requests' },
           () => router.refresh()
         )
         .subscribe()
       return () => {
         client.removeChannel(channel)
       }
     }, [router])

     async function handleAccept(swap: IncomingSwapRequest) {
       const accepterTaskId = selectedMap[swap.id] ?? swap.my_matching_task_ids[0]
       if (!accepterTaskId) return
       setPending((p) => new Set(p).add(swap.id))
       setErrors((e) => ({ ...e, [swap.id]: '' }))
       try {
         await acceptSwapRequest(swap.id, accepterTaskId)
         // Optimistic removal; router.refresh will also land
         setIncoming((rows) => rows.filter((r) => r.id !== swap.id))
         router.refresh()
       } catch (err) {
         setErrors((e) => ({
           ...e,
           [swap.id]: err instanceof Error ? err.message : 'Ukendt fejl',
         }))
       } finally {
         setPending((p) => {
           const next = new Set(p)
           next.delete(swap.id)
           return next
         })
       }
     }

     if (incoming.length === 0) return null

     return (
       <section className="space-y-3">
         <h2 className="text-base font-semibold">Indkommende bytte-forespørgsler</h2>
         <ul className="space-y-2">
           {incoming.map((swap) => {
             const options = swap.my_matching_task_ids
             const selected = selectedMap[swap.id] ?? options[0]
             const isPending = pending.has(swap.id)
             const err = errors[swap.id]
             return (
               <li key={swap.id} className="rounded-md border p-3 space-y-2">
                 <div className="text-sm">
                   <p><span className="font-medium">{swap.requester_name}</span> vil bytte <span className="italic">{swap.requester_task_title}</span> til en af dine opgaver.</p>
                 </div>

                 {options.length > 1 && (
                   <div>
                     <label className="text-xs text-muted-foreground" htmlFor={`target-${swap.id}`}>Vælg opgave at bytte med:</label>
                     <select
                       id={`target-${swap.id}`}
                       value={selected}
                       onChange={(e) =>
                         setSelectedMap((m) => ({ ...m, [swap.id]: e.target.value }))
                       }
                       className="flex h-9 w-full rounded-md border border-input bg-background px-3 py-1 text-sm"
                     >
                       {options.map((taskId) => (
                         <option key={taskId} value={taskId}>
                           {taskTitleMap[taskId] ?? taskId}
                         </option>
                       ))}
                     </select>
                   </div>
                 )}
                 {options.length === 1 && (
                   <p className="text-xs text-muted-foreground">Du bytter med: {taskTitleMap[options[0]] ?? options[0]}</p>
                 )}

                 {err && <p className="text-xs text-destructive">{err}</p>}

                 <div className="flex gap-2">
                   <Button type="button" size="sm" disabled={isPending} onClick={() => handleAccept(swap)}>
                     {isPending ? 'Accepterer…' : 'Acceptér bytte'}
                   </Button>
                 </div>
               </li>
             )
           })}
         </ul>
       </section>
     )
   }
   ```
